germanic_conversion_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
    }

    desc = germanic_conversion_decision_desc
    selection_tooltip = germanic_conversion_decision_tooltip
    confirm_text = germanic_conversion_decision_confirm

    # This decision should be visible to Christian West Germanic rulers in Britannia
    is_shown = {
        culture = { has_cultural_pillar = heritage_west_germanic }
        religion = religion:christianity_religion
        is_landed = yes
		any_held_title = {
			title_tier = county
			title_province = { geographical_region = world_europe_west_britannia }
		}
    }

    is_valid = {
        # West Germanic Heritage
        culture = { has_cultural_pillar = heritage_west_germanic }

        # Christian Religion
        religion = religion:christianity_religion

        # Landed Ruler
        is_landed = yes

        # Capital county with West Germanic cultural heritage
        custom_tooltip = {
            text = germanic_conversion_county_culture_req_tooltip
            capital_county.culture = {
                has_cultural_pillar = heritage_west_germanic
            }
        }

        # Primary title within England or Scotland
        custom_tooltip = {
            text = germanic_conversion_location_req_tooltip

            OR = {
                # County within England or Scotland
                primary_title = {
                    de_jure_liege = {
                        OR = {
                            de_jure_liege = title:k_england
                            de_jure_liege = title:k_scotland
                        }
                    }
                }

                # Duchy within England or Scotland
                primary_title = {
                    OR = {
                        de_jure_liege = title:k_england
                        de_jure_liege = title:k_scotland
                    }
                }
            }
        }
    }

    cost = {
        piety = 150
        prestige = 150
    }

    effect = {
        # The ruler converts to saxon paganism
        set_character_faith = faith:west_germanic_pagan

        # Every non-zealous vassal converts
        every_vassal = {
            limit = {
                NOT = { has_trait = zealous}
            }
            set_character_faith = faith:west_germanic_pagan
        }

        # All non-zealous courtiers will convert as well
        every_courtier = {
            limit = {
                NOT = { has_trait = zealous }
            }
            set_character_faith = faith:west_germanic_pagan
        }

        # Convert the faith of the ruler's capital county, and apply a positive county modifier
        capital_county = { 
            set_county_faith = faith:west_germanic_pagan 
            add_county_modifier = {
                modifier = germanic_conversion_county_modifier
                years = 10
            }                    
        }

        # Grant the ruler a stack of special troops
        custom_tooltip = {
            text = germanic_conversion_event_troops_description

            spawn_army = {
                men_at_arms = {
                    type = gesithas
                    stacks = 1
                }
                men_at_arms = {
                    type = fyrd_spearmen
                    stacks = 2
                }
                men_at_arms = {
                    type = fyrd_skirmishers
                    stacks = 2
                }
                location = capital_province
                name = germanic_conversion_event_troops
            }
        }

        # Present an event panel for the player
        trigger_event = germanic_conversion_event.0001
    }


    ai_potential = {
        # AI rulers must be an independent count, or a duke
        OR = {
            # Independent ruler with count-tier primary title, or
            AND = {
                primary_title.tier = tier_county
                is_independent_ruler = yes
            }

            # Ruler with duke-tier primary title
            primary_title.tier = tier_duchy
        }

        # AI rulers must not be Zealous to consider this decision
        ai_zeal < high_positive_ai_value                
    }

    ai_will_do = {
        # Begin with a relatively low base chance
        base = 10

        # AI rulers are significantly less likely to take this decision in the late medieval era
        modifier = {
            factor = 0.25
            culture = { has_cultural_era_or_later = culture_era_late_medieval }
        }

        # AI rulers are less likely to take this decision in the high medieval era
        modifier = {
            factor = 0.5
            culture = { has_cultural_era_or_later = culture_era_high_medieval }
        }

        # AI rulers are slightly less likely to take this decision in the early medieval era
        modifier = {
            factor = 0.75
            culture = { has_cultural_era_or_later = culture_era_early_medieval }
        }

        # AI rulers are more likely to take this decision in the tribal era
        modifier = {
            factor = 1.5
            culture = { has_cultural_era_or_later = culture_era_tribal }
        }

        # Characters whose liege or suzerain follow a Germanic faith are more likely to convert
        modifier = {
            factor = 4
            OR = {
                liege = {
                    religion = religion:germanic_religion
                }
                suzerain = {
                    religion = religion:germanic_religion
                }
            }
        }

        # Cowardly characters are unlikely to convert
        modifier = {
            factor = 0.1
            ai_boldness < medium_negative_ai_value
        }

        # Bold characters are more likely to convert
        modifier = {
            factor = 3
            AND = {
                ai_boldness >= medium_positive_ai_value
                ai_boldness < high_positive_ai_value
            }
        }

        # Fearless characters are much more likely to convert
        modifier = {
            factor = 5
            ai_boldness >= high_positive_ai_value
        }

        # Compassionate characters are less likely to convert
        modifier = {
            factor = 0.25
            ai_compassion >= low_positive_ai_value
        }

        # Content characters are less likely to convert
        modifier = {
            factor = 0.25
            ai_greed < low_negative_ai_value
        }

        # Covetous characters are more likely to convert
        modifier = {
            factor = 3
            AND = {
                ai_greed >= medium_positive_ai_value
                ai_greed < high_positive_ai_value
            }
        }

        # Rapacious characters are much more likely to convert
        modifier = {
            factor = 5
            ai_greed >= high_positive_ai_value
        }

        # Forgiving characters are less likely to convert
        modifier = {
            factor = 0.25
            ai_vengefulness < low_negative_ai_value
        }

        # Resentful characters are more likely to convert
        modifier = {
            factor = 3
            AND = {
                ai_vengefulness >= medium_positive_ai_value
                ai_vengefulness < high_positive_ai_value
            }
        }

        # Vindictive characters are much more likely to convert
        modifier = {
            factor = 5
            ai_vengefulness >= high_positive_ai_value
        }

        # Zealous characters are highly unlikely to convert
        modifier = {
            factor = 0.1
            ai_zeal >= medium_positive_ai_value
        }
    }

    
    # The AI should review this decision every 5 years
    ai_check_interval = 60
}
