west_germanic_conversion_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
    }

    desc = west_germanic_conversion_decision_desc
    selection_tooltip = west_germanic_conversion_decision_tooltip
    confirm_text = west_germanic_conversion_decision_confirm

    # This decision should be visible to Christian West Germanic rulers in England or Scotland during the Tribal Era
    is_shown = {
        culture = { has_cultural_pillar = heritage_west_germanic }
        religion = religion:christianity_religion
        is_landed = yes
		any_held_title = {
			title_tier = county
			title_province = { geographical_region = world_europe_west_britannia }
		}
		culture = { NOT = { has_cultural_era_or_later = culture_era_early_medieval } }
    }

    is_valid = {
        # West Germanic Heritage
        culture = { has_cultural_pillar = heritage_west_germanic }

        # Christian Religion
        religion = religion:christianity_religion

        # Landed Ruler
        is_landed = yes

        # Primary title within England or Scotland
        custom_tooltip = {
            text = west_germanic_conversion_location_req_tooltip

            OR = {
                # County within England or Scotland
                primary_title = {
                    de_jure_liege = {
                        OR = {
                            de_jure_liege = title:k_england
                            de_jure_liege = title:k_scotland
                        }
                    }
                }

                # Duchy within England or Scotland
                primary_title = {
                    OR = {
                        de_jure_liege = title:k_england
                        de_jure_liege = title:k_scotland
                    }
                }
            }
        }

        # Tribal Era
		culture = { NOT = { has_cultural_era_or_later = culture_era_early_medieval } }
    }

    cost = {
        piety = 150
        prestige = 150
    }

    effect = {
        # The ruler converts to Asatru
        set_character_faith = faith:norse_pagan

        # All non-zealous courtiers will convert as well
        every_courtier = {
            limit = {
                NOT = { has_trait = zealous }
            }
            set_character_faith = faith:norse_pagan
        }

        # Convert the faith of the ruler's capital county
        capital_county = { 
            set_county_faith = faith:norse_pagan 
        }

        # Present an event panel for the player
        trigger_event = west_germanic_conversion.0001
    }

    cooldown = { years = 10 }


    ai_potential = {
        OR = {
            # Independent ruler with count-tier primary title, or
            AND = {
                primary_title.tier = tier_county
                is_independent_ruler = yes
            }

            # Ruler with duke-tier primary title
            primary_title.tier = tier_duchy
        }

        # Non-zealous
        NOT = { 
            has_trait = zealous 
        }
    }

    ai_will_do = {
        base = 20

        # Brave, Wrathful, Ambitious, Vengeful, Greedy and Cynical rulers are more likely to take this decision
        modifier = {
            factor = 2
            has_trait = brave
        }

        modifier = {
            factor = 2
            has_trait = wrathful
        }

        modifier = {
            factor = 2
            has_trait = ambitious
        }

        modifier = {
            factor = 2
            has_trait = vengeful
        }

        modifier = {
            factor = 2
            has_trait = greedy
        }

        modifier = {
            factor = 2
            has_trait = cynical
        }

        # Whereas Craven, Content, Humble, Compassionate, and Forgiving rulers are less likely to take the decision
        modifier = {
            factor = 0.5
            has_trait = craven
        }

        modifier = {
            factor = 0.5
            has_trait = content
        }

        modifier = {
            factor = 0.5
            has_trait = humble
        }

        modifier = {
            factor = 0.5
            has_trait = compassionate
        }

        modifier = {
            factor = 0.5
            has_trait = forgiving
        }
    }

    
    # The AI should check this decision every 24 months
    ai_check_interval = 24
}
