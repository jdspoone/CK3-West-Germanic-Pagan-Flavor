welsh_longbow_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_castle_view.dds"
    }

    desc = welsh_longbow_decision_desc
    selection_tooltip = welsh_longbow_decision_tooltip
    confirm_text = welsh_longbow_decision_confirm


    is_shown = {
        # Landed Ruler
        is_landed = yes
        
        # West Germanic cultural heritage
        culture = {  has_cultural_pillar = heritage_west_germanic  }

        # Culture doesn't already have Longbow Competitions cultural tradition
        NOT = { culture = { has_cultural_tradition = tradition_longbow_competitions } }
    }

    is_valid = {
        # Landed Ruler
        is_landed = yes

        # West Germanic cultural heritage
        culture = {  has_cultural_pillar = heritage_west_germanic  }

        # Moderately prestigious
		prestige_level >= 3

        # Culture head
        custom_tooltip = {
            text = welsh_longbow_decision_culture_head_requirement
            culture = { culture_head = root }
        }

        # Cultural acceptence with Welsh >= 50
        culture = {
            cultural_acceptance = {
                target = culture:welsh
                value >= 50
            }
        }

        # Requires having a friend who is a Welsh ruler within Wales
        custom_tooltip = {
            text = welsh_longbow_decision_friend_requirement

            any_relation = {
                type = friend
                culture = culture:welsh
                is_landed = yes

                OR = {
                    # County within Wales
                    primary_title = {
                        de_jure_liege = {
                            de_jure_liege = title:k_wales
                        }
                    }

                    # Duchy within Wales
                    primary_title = {
                        de_jure_liege = title:k_wales
                    }

                    # Kingdom of Wales
                    primary_title = title:k_wales
                }
            }
        }
    }

    cost = {
        prestige = 500
        gold = 250
    }


    effect = {
        culture = { add_culture_tradition = tradition_longbow_competitions }

        # Present an event panel for the player
        trigger_event = welsh_longbow_event.0001
    }


    ai_check_interval = 0    
}
