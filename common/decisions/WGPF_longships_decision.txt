longships_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
    }

    desc = longships_decision_desc
    selection_tooltip = longships_decision_tooltip
    confirm_text = longships_decision_confirm


    is_shown = {
        # West Germanic cultural heritage
        culture = { 
            has_cultural_pillar = heritage_west_germanic
        }

        # Culture does not already have the longboats innovation
        NOT = { 
            culture = { 
                has_innovation = innovation_longboats 
            }
        }
    }

    is_valid = {
        # Landed Ruler
        is_landed = yes

        # West Germanic cultural heritage
        culture = {  has_cultural_pillar = heritage_west_germanic  }

        # Moderately prestigious
		prestige_level >= 3

        # Friends with a North Germanic ruler following the same religious family as you
        custom_tooltip = {
            text = longships_decision_friend_req
            any_relation = {
                type = friend
                is_landed = yes
                culture = {  
                    has_cultural_pillar = heritage_north_germanic 
                }
                religion = root.religion
            }
        }
    }

    cost = {
        prestige = 500
        gold = 125
    }


    effect = {
        culture = { add_innovation = innovation_longboats }

        # Present an event panel for the player
        trigger_event = longships_event.0001
    }


    ai_check_interval = 0    
}
