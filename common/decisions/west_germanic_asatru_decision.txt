west_germanic_to_asatru_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
    }

    desc = west_germanic_to_asatru_decision_desc
    selection_tooltip = west_germanic_to_asatru_decision_tooltip
    confirm_text = west_germanic_to_asatru_decision_confirm

	decision_group_type = minor

    # This decision should be visible to Christian West Germanic rulers in England during the Tribal Era
    is_shown = {
        culture = { has_cultural_pillar = heritage_west_germanic }
        religion = religion:christianity_religion
        is_landed = yes
		any_held_title = {
			title_tier = county
			title_province = { geographical_region = world_europe_west_britannia }
		}
		culture = { NOT = { has_cultural_era_or_later = culture_era_early_medieval } }
    }

    # This decision is only available for West Germanic Christians whose primary title is a duchy in England during the Tribal era
    is_valid = {
        culture = { has_cultural_pillar = heritage_west_germanic }
        religion = religion:christianity_religion
        is_landed = yes
        primary_title = {
            de_jure_liege = title:k_england
        }
		culture = { NOT = { has_cultural_era_or_later = culture_era_early_medieval } }
    }

    cost = {
        piety = 150
        prestige = 150
    }

    effect = {
        # The ruler converts to Asatru
        set_character_faith = faith:norse_pagan

        # All non-zealous courtiers will convert as well
        every_courtier = {
            limit = {
                NOT = { has_trait = zealous }
            }
            set_character_faith = faith:norse_pagan
        }
    }

    cooldown = { years = 10 }

    ai_potential = {
        # Limit the decision to non-zealous AI rulers
        NOT = { has_trait = zealous }
    }

    ai_will_do = {
        base = 20

        # Brave, Wrathful, Ambitious, Vengeful, Greedy and Cynical rulers are more likely to take this decision
        modifier = {
            factor = 2
            has_trait = brave
        }

        modifier = {
            factor = 2
            has_trait = wrathful
        }

        modifier = {
            factor = 2
            has_trait = ambitious
        }

        modifier = {
            factor = 2
            has_trait = vengeful
        }

        modifier = {
            factor = 2
            has_trait = greedy
        }

        modifier = {
            factor = 2
            has_trait = cynical
        }

        # Whereas Craven, Content, Humble, Compassionate, and Forgiving rulers are less likely to take the decision
        modifier = {
            factor = 0.5
            has_trait = craven
        }

        modifier = {
            factor = 0.5
            has_trait = content
        }

        modifier = {
            factor = 0.5
            has_trait = humble
        }

        modifier = {
            factor = 0.5
            has_trait = compassionate
        }

        modifier = {
            factor = 0.5
            has_trait = forgiving
        }
    }

    
    # The AI should check this decision every 24 months
    ai_check_interval = 24
}
